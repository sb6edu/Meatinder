KUNDEN_Tabelle
CREATE TABLE KUNDEN (KUNID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, VORNAME VARCHAR(30) NOT NULL, NACHNAME VARCHAR(30) NOT NULL, USERNAME VARCHAR(20) NOT NULL, EMAIL VARCHAR(50) NOT NULL, PASSWORT VARCHAR(20) NOT NULL, PRIMARY KEY (KUNID));

Artikel
CREATE TABLE ARTIKEL (ARTID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, ARTNAME VARCHAR(30), PRIMARY KEY (ARTID));

Rezeptartikel
CREATE TABLE REZEPTARTIKEL (RID INTEGER NOT NULL, ARTID INTEGER NOT NULL, MENGE INTEGER NOT NULL, EINHEIT VARCHAR(15) NOT NULL, PRIMARY KEY (RID, ARTID));

Rezepte
CREATE TABLE REZEPTE (ID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, REZEPTNAME VARCHAR(50), ZUBEREITUNGSZEIT VARCHAR(20), PRIMARY KEY (ID));
